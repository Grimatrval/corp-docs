require('dotenv').config(); const { Telegraf, Markup } = require('telegraf'); const { Pool } = require('pg'); const bot = new Telegraf(process.env.BOT_TOKEN); const pool = new Pool({ connectionString: process.env.DATABASE_URL }); const userStates = {}; // ========== КОМАНДЫ ========== bot.start(async (ctx) => { const userId = ctx.from.id.toString(); let user = await pool.query('SELECT * FROM users WHERE telegram_id = $1', [userId]); if (user.rows.length === 0) { await pool.query('INSERT INTO users (telegram_id, username, first_name) VALUES ($1, $2, $3)', [userId, ctx.from.username, ctx.from.first_name]); } ctx.reply(`👋 *Добро пожаловать, ${ctx.from.first_name}!*\n\n📋 *Корпоративный Документооборот*\n\n*Команды:*\n/newapproval — Создать согласование\n/newtask — Создать поручение\n/mytasks — Мои задачи\n/myerrands — Мои поручения\n/myapprovals — Мои согласования\n/help — Помощь\n\n*Админ:*\n/register — Добавить пользователя\n/unregister — Удалить пользователя\n/listusers — Список\n\n*Или отправьте файл* с текстом "Согласование: ..."\n\nВы: @${ctx.from.username || 'нет'}\nID: ${ctx.from.id}`, { parse_mode: 'Markdown' }); }); bot.help((ctx) => { ctx.reply(`📖 *Помощь*\n\n*Создание:*\n/newapproval — согласование\n/newtask — поручение\n\n*Просмотр:*\n/mytasks — мои задачи\n/myerrands — созданные мной поручения\n/myapprovals — мои согласования\n\n*Файлы:*\nОтправьте файл с текстом "Согласование: название"`, { parse_mode: 'Markdown' }); }); // ========== АДМИН КОМАНДЫ ========== bot.command('register', async (ctx) => { const user = await pool.query('SELECT * FROM users WHERE telegram_id = $1', [ctx.from.id.toString()]); if (user.rows.length === 0 || user.rows[0].role !== 'admin') return ctx.reply('❌ Только админ'); userStates[ctx.from.id] = { step: 'reg_user' }; ctx.reply('👤 Введите @username:', { reply_markup: Markup.keyboard([['❌ Отмена']]).resize().oneTime() }); }); bot.command('unregister', async (ctx) => { const user = await pool.query('SELECT * FROM users WHERE telegram_id = $1', [ctx.from.id.toString()]); if (user.rows.length === 0 || user.rows[0].role !== 'admin') return ctx.reply('❌ Только админ'); userStates[ctx.from.id] = { step: 'unreg_user' }; ctx.reply('👤 Введите @username для удаления:', { reply_markup: Markup.keyboard([['❌ Отмена']]).resize().oneTime() }); }); bot.command('listusers', async (ctx) => { const user = await pool.query('SELECT * FROM users WHERE telegram_id = $1', [ctx.from.id.toString()]); if (user.rows.length === 0 || user.rows[0].role !== 'admin') return ctx.reply('❌ Только админ'); const result = await pool.query('SELECT username, role, is_active FROM users ORDER BY username'); let msg = '👥 *Пользователи:*\n\n'; result.rows.forEach((u, i) => { msg += `${i+1}. ${u.is_active ? '✅' : '❌'} @${u.username || 'нет'} — ${u.role}\n`; }); ctx.reply(msg, { parse_mode: 'Markdown' }); }); // ========== СОГЛАСОВАНИЯ ========== bot.command('newapproval', (ctx) => { userStates[ctx.from.id] = { step: 'appr_title' }; ctx.reply('📄 Название:', { reply_markup: Markup.keyboard([['❌ Отмена']]).resize().oneTime() }); }); bot.command('myapprovals', async (ctx) => { const user = await pool.query('SELECT id FROM users WHERE telegram_id = $1', [ctx.from.id.toString()]); if (user.rows.length === 0) return ctx.reply('❌ Не найден'); const result = await pool.query('SELECT * FROM approvals WHERE creator_id = $1 ORDER BY created_at DESC LIMIT 20', [user.rows[0].id]); if (result.rows.length === 0) return ctx.reply('📭 Нет согласований'); let msg = '📋 *Ваши согласования:*\n\n'; result.rows.forEach((a, i) => { msg += `${i+1}. *${a.title}* — ${a.amount}₽ | @${a.approver_username || 'н/д'} | ${a.status}\n`; }); ctx.reply(msg, { parse_mode: 'Markdown' }); }); // ========== ПОРУЧЕНИЯ ========== bot.command('newtask', (ctx) => { userStates[ctx.from.id] = { step: 'task_title' }; ctx.reply('✅ Название задачи:', { reply_markup: Markup.keyboard([['❌ Отмена']]).resize().oneTime() }); }); bot.command('mytasks', async (ctx) => { const user = await pool.query('SELECT * FROM users WHERE telegram_id = $1', [ctx.from.id.toString()]); if (user.rows.length === 0) return ctx.reply('❌ Не найден'); const result = await pool.query("SELECT * FROM tasks WHERE executor_username = $1 AND status != 'completed' ORDER BY deadline ASC", [user.rows[0].username]); if (result.rows.length === 0) return ctx.reply('📭 Нет задач'); let msg = '📋 *Ваши задачи:*\n\n'; result.rows.forEach((t, i) => { msg += `${i+1}. *${t.title}*\n📅 ${new Date(t.deadline).toLocaleDateString('ru-RU')} | 🔥 ${t.priority}\n\n`; }); ctx.reply(msg, { parse_mode: 'Markdown' }); }); bot.command('myerrands', async (ctx) => { const user = await pool.query('SELECT id FROM users WHERE telegram_id = $1', [ctx.from.id.toString()]); if (user.rows.length === 0) return ctx.reply('❌ Не найден'); const result = await pool.query('SELECT * FROM tasks WHERE creator_id = $1 ORDER BY created_at DESC LIMIT 20', [user.rows[0].id]); if (result.rows.length === 0) return ctx.reply('📭 Нет поручений'); let msg = '📝 *Ваши поручения:*\n\n'; result.rows.forEach((t, i) => { msg += `${i+1}. *${t.title}*\n👤 @${t.executor_username} | 📅 ${new Date(t.deadline).toLocaleDateString('ru-RU')}\n\n`; }); ctx.reply(msg, { parse_mode: 'Markdown' }); }); // ========== ОБРАБОТКА ТЕКСТА ========== bot.on('text', async (ctx) => { const text = ctx.message.text; const userId = ctx.from.id; if (text === '❌ Отмена') { delete userStates[userId]; return ctx.reply('❌ Отменено', { reply_markup: Markup.removeKeyboard() }); } // Авто-создание согласования if (text.toLowerCase().includes('согласование:') || text.toLowerCase().includes('согласуй:')) { const title = text.replace(/согласование:|согласуй:/i, '').trim(); const user = await pool.query('SELECT id FROM users WHERE telegram_id = $1', [userId.toString()]); let creatorId; if (user.rows.length === 0) { const newUser = await pool.query('INSERT INTO users (telegram_id, username, first_name) VALUES ($1, $2, $3) RETURNING id', [userId.toString(), ctx.from.username, ctx.from.first_name]); creatorId = newUser.rows[0].id; } else creatorId = user.rows[0].id; const result = await pool.query("INSERT INTO approvals (title, description, creator_id, status) VALUES ($1, $2, $3, 'pending') RETURNING *", [title, 'Авто-создание', creatorId]); return ctx.reply(`✅ Согласование #${result.rows[0].id} создано!\n\n📄 ${title}`); } const state = userStates[userId]; // Регистрация if (state?.step === 'reg_user') { const username = text.replace('@', '').trim(); userStates[userId] = { ...state, username, step: 'reg_role' }; ctx.reply('🔑 Роль (employee/approver/admin):'); return; } if (state?.step === 'reg_role') { const currentUser = await pool.query('SELECT * FROM users WHERE telegram_id = $1', [userId.toString()]); if (currentUser.rows.length === 0 || currentUser.rows[0].role !== 'admin') { delete userStates[userId]; return ctx.reply('❌ Только админ'); } await pool.query('INSERT INTO users (telegram_id, username, role) VALUES ($1, $2, $3) ON CONFLICT (telegram_id) DO UPDATE SET role = $3', [`temp_${state.username}`, state.username, state.role || text]); delete userStates[userId]; return ctx.reply(`✅ @${state.username} добавлен!`); } // Удаление if (state?.step === 'unreg_user') { const username = text.replace('@', '').trim(); await pool.query("UPDATE users SET is_active = false WHERE username = $1", [username]); delete userStates[userId]; return ctx.reply(`✅ @${username} удалён`); } // Согласование if (state?.step === 'appr_title') { userStates[userId] = { ...state, title: text, step: 'appr_amount' }; return ctx.reply('💰 Сумма:'); } if (state?.step === 'appr_amount') { userStates[userId] = { ...state, amount: parseFloat(text) || 0, step: 'appr_desc' }; return ctx.reply('📝 Описание:'); } if (state?.step === 'appr_desc') { userStates[userId] = { ...state, description: text, step: 'appr_approver' }; return ctx.reply('👤 @username согласующего:'); } if (state?.step === 'appr_approver') { const approver = text.replace('@', '').trim(); const user = await pool.query('SELECT id FROM users WHERE telegram_id = $1', [userId.toString()]); let creatorId; if (user.rows.length === 0) { const newUser = await pool.query('INSERT INTO users (telegram_id, username, first_name) VALUES ($1, $2, $3) RETURNING id', [userId.toString(), ctx.from.username, ctx.from.first_name]); creatorId = newUser.rows[0].id; } else creatorId = user.rows[0].id; const result = await pool.query("INSERT INTO approvals (title, description, amount, creator_id, approver_username, status) VALUES ($1, $2, $3, $4, $5, 'pending') RETURNING *", [state.title, state.description, state.amount, creatorId, approver]); delete userStates[userId]; ctx.reply(`✅ Согласование #${result.rows[0].id} создано!\n\n📄 ${state.title}\n💰 ${state.amount}₽\n👤 @${approver}`); // Уведомление const approverUser = await pool.query('SELECT telegram_id FROM users WHERE username = $1', [approver]); if (approverUser.rows.length > 0) { await bot.telegram.sendMessage(approverUser.rows[0].telegram_id, `🔔 Новое согласование #${result.rows[0].id}\n\n📄 ${state.title}\n💰 ${state.amount}₽\n\nОт: @${ctx.from.username || ctx.from.first_name}`); } return; } // Поручение if (state?.step === 'task_title') { userStates[userId] = { ...state, title: text, step: 'task_desc' }; return ctx.reply('📝 Описание:'); } if (state?.step === 'task_desc') { userStates[userId] = { ...state, description: text, step: 'task_exec' }; return ctx.reply('👤 @username исполнителя:'); } if (state?.step === 'task_exec') { userStates[userId] = { ...state, executor: text.replace('@', '').trim(), step: 'task_deadline' }; return ctx.reply('📅 Срок (ДД.ММ.ГГГГ):'); } if (state?.step === 'task_deadline') { userStates[userId] = { ...state, deadline: text, step: 'task_priority' }; ctx.reply('🔥 Приоритет:', { reply_markup: Markup.keyboard([['Низкий'], ['Средний'], ['Высокий'], ['❌ Отмена']]).resize().oneTime() }); return; } if (state?.step === 'task_priority') { const priority = { 'Низкий': 'low', 'Средний': 'medium', 'Высокий': 'high' }[text] || 'medium'; const user = await pool.query('SELECT id FROM users WHERE telegram_id = $1', [userId.toString()]); let creatorId; if (user.rows.length === 0) { const newUser = await pool.query('INSERT INTO users (telegram_id, username, first_name) VALUES ($1, $2, $3) RETURNING id', [userId.toString(), ctx.from.username, ctx.from.first_name]); creatorId = newUser.rows[0].id; } else creatorId = user.rows[0].id; const parts = state.deadline.split('.'); const deadline = `${parts[2]}-${parts[1]}-${parts[0]}`; const result = await pool.query("INSERT INTO tasks (title, description, creator_id, executor_username, deadline, priority, status) VALUES ($1, $2, $3, $4, $5, $6, 'pending') RETURNING *", [state.title, state.description, creatorId, state.executor, deadline, priority]); delete userStates[userId]; ctx.reply(`✅ Поручение #${result.rows[0].id} создано!\n\n📋 ${state.title}\n👤 @${state.executor}\n📅 ${state.deadline}\n🔥 ${text}`); // Уведомление const execUser = await pool.query('SELECT telegram_id FROM users WHERE username = $1', [state.executor]); if (execUser.rows.length > 0) { await bot.telegram.sendMessage(execUser.rows[0].telegram_id, `🔔 Новое поручение #${result.rows[0].id}\n\n📋 ${state.title}\n📝 ${state.description}\n📅 ${state.deadline}\n\nОт: @${ctx.from.username || ctx.from.first_name}`); } } }); // ========== ЗАПУСК ========== bot.launch().then(() => console.log('✅ Бот запущен!')); process.on('SIGINT', () => bot.stop()); process.on('SIGTERM', () => bot.stop());
